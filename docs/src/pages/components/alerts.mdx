    }
    AlertHeading: componentMetadata(displayName: { eq: "AlertHeading" }) {
      ...ComponentApi_metadata
    }
    AlertLink: componentMetadata(displayName: { eq: "AlertLink" }) {
      ...ComponentApi_metadata
    }
    AlertDismissible: componentMetadata(
      displayName: { eq: "AlertDismissible" }
    ) {
      ...ComponentApi_metadata
    }
  }
